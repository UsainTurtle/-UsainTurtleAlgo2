# 과제는 끝나지 않아(bj 17952)

## 문제 풀이 과정

1. map에 입력값을 입력받은 후 0~3까지 중복순열을 만든다.
2. 중복순열 결과를 arr에 저장해 0~3까지 각 4방향에 맞는 set메서드를 호출한다.
3. boolean형 2차원 배열을 만들어 한번 합쳐진 위치를 체크한다.
4. map 전체에서 가장 큰 값을 max에 저장한다.
5. 전체 중복순열을 탐색하여 max값을 출력한다.

## 전체 코드

```java
import java.io.*;
import java.util.*;
public class Main {
	static int N, max=0;
	public static void main(String[] args) throws Exception{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		int[][] map = new int[N][N];
		for(int i=0; i<N; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine()," ");
			for(int j=0; j<N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}
		int[] arr = new int[5];
		p(map,arr,0);
		System.out.println(max);
		br.close();
	}
	static void p(int[][] map, int[] arr, int idx) {
		if(idx==5) {
			st_game(copy_map(map),arr);
			return;
		}
		for(int i=0; i<4; i++) {
			arr[idx]=i;
			p(map,arr,idx+1);
		}
	}
	static void st_game(int[][]map, int[] arr) {
		for(int i=0; i<arr.length; i++) {
			switch(arr[i]) {
			case 0:
				set0(map, new boolean[N][N]);
				break;
			case 1:
				set1(map, new boolean[N][N]);
				break;
			case 2:
				set2(map, new boolean[N][N]);
				break;
			case 3:
				set3(map, new boolean[N][N]);
				break;
			}
		}
		max = Math.max(sum(map),max);
	}
	static int[][] copy_map(int[][] map) {
		int[][] tmp = new int[N][N];
		for(int i=0; i<N; i++) {
			for(int j=0; j<N; j++) {
				tmp[i][j]=map[i][j];
			}
		}
		return tmp;
	}
	static void set0(int[][] map, boolean[][] v) {
		for(int i=0; i<N; i++) {
			for(int j=1; j<N; j++) {
				int idx=j;
				while(true) {
					idx--;
					if(idx<0) {
						map[i][0]=map[i][j];
						if(0!=j) map[i][j]=0;
						break;
					}
					if(map[i][idx]!=0) {
						if(!v[i][idx] && map[i][j]==map[i][idx]) {
							v[i][idx]=true;
							map[i][idx]=map[i][j]+map[i][idx];
							map[i][j]=0;
						}
						else {
							map[i][++idx]=map[i][j];
							if(idx!=j) map[i][j]=0;
						}
						break;
					}
				}
			}
		}
	}
	static void set1(int[][] map, boolean[][] v) {
		for(int j=0; j<N; j++) {
			for(int i=N-2; 0<=i; i--) {
				int idx=i;
				while(true) {
					idx++;
					if(N<=idx) {
						map[N-1][j]=map[i][j];
						if(N-1!=i) map[i][j]=0;
						break;
					}
					if(map[idx][j]!=0) {
						if(!v[idx][j] && map[i][j]==map[idx][j]) {
							v[idx][j]=true;
							map[idx][j]=map[i][j]+map[idx][j];
							map[i][j]=0;
						}
						else {
							map[--idx][j]=map[i][j];
							if(idx!=i) map[i][j]=0;
						}
						break;
					}
				}
			}
		}
	}
	static void set2(int[][] map, boolean[][] v) {
		for(int i=0; i<N; i++) {
			for(int j=N-2; 0<=j; j--) {
				int idx=j;
				while(true) {
					idx++;
					if(N<=idx) {
						map[i][N-1]=map[i][j];
						if(N-1!=j) map[i][j]=0;
						break;
					}
					if(map[i][idx]!=0) {
						if(!v[i][idx] && map[i][j]==map[i][idx]) {
							v[i][idx]=true;
							map[i][idx]=map[i][j]+map[i][idx];
							map[i][j]=0;
						}
						else {
							map[i][--idx]=map[i][j];
							if(idx!=j) map[i][j]=0;
						}
						break;
					}
				}
			}
		}
	}
	static void set3(int[][] map, boolean[][] v) {
		for(int j=0; j<N; j++) {
			for(int i=1; i<N; i++) {
				int idx=i;
				while(true) {
					idx--;
					if(idx<0) {
						map[0][j]=map[i][j];
						if(0!=i) map[i][j]=0;
						break;
					}
					if(map[idx][j]!=0) {
						if(!v[idx][j] &&map[i][j]==map[idx][j]) {
							v[idx][j]=true;
							map[idx][j]=map[i][j]+map[idx][j];
							map[i][j]=0;
						}
						else {
							map[++idx][j]=map[i][j];
							if(idx!=i) map[i][j]=0;
						}
						break;
					}
				}
			}
		}
	}
	static int sum(int[][] map) {
		int max=0;
		for(int i=0; i<N; i++) {
			for(int j=0; j<N; j++) {
				max=Math.max(max,map[i][j]);
			}
		}
		return max;
	}
}
```
